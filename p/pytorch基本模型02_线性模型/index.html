import numpy as np
import matplotlib.pyplot as plt

x_data = [1.0,2.0,3.0]
y_data = [2.0,4.0,6.0]

#正向传播，预测y
def forward(x):
    return x*w

#计算损失
def loss(x,y):
    y_pred = forward(x)
    return (y_pred-y) * (y_pred-y)

#权重更新存储
w_list = []
#mse均误差cunc
mse_list = []
#产生不同的权重，用于计算  左闭右开 
for w in np.arange(0.0,4.1,0.1):
    print("w=",w)
    l_sum = 0
    #依次传入每个样本
    for x_val,y_val in zip(x_data,y_data):
        y_pred_val = forward(x_val)
        #注意定义的loss中本身会计算产生一个正向传播的计算图
        loss_val = loss(x_val,y_val)
        #mse这里是三个样本的损失之和
        l_sum+=loss_val
        #t制表符会使得输出更好看
        print("\t",x_val,y_val,y_pred_val,loss_val)
    print("MSE=",l_sum/3)
    w_list.append(w)
    mse_list.append(l_sum/3)
