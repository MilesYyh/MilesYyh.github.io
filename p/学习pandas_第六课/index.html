#创建一个groupby对象
one = df.groupby("letter")#根据letter分组   即根据letter的结果分组，相同的放一起
one.sum()


#多个分组依据记得[]封起来
letterone = df.groupby(["letter","one"]).sum()
letterone


letterone.index

result:

你可能不想把用来分组的列名字作为索引，像下面的做法很容易实现。

#参数as_index=False会取消把groupby的对象作为index
letterone = df.groupby(["letter","one"],as_index=False).sum()
letterone
