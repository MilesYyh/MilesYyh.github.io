#可以将上面这个封装成一个方法去调用
def plot_learning_curve(algo,X_train,X_test,y_train,y_test):
    """绘制学习曲线：只需要传入算法(或实例对象)、X_train、X_test、y_train、y_test"""
    """当使用该函数时传入算法，该算法的变量要进行实例化，如：PolynomialRegression(degree=2)，变量 degree 要进
    行实例化"""
    train_score = []
    test_score = []
    for i in range(10,len(X_train)+1,10):
        algo.fit(X_train[:i],y_train[:i])
        
        y_train_predict = algo.predict(X_train[:i])
        train_score.append(mean_squared_error(y_train_predict,y_train[:i]))
        
        y_test_predict = algo.predict(X_test)
        test_score.append(mean_squared_error(y_test_predict,y_test))
        
    plt.plot([i for i in range(1,len(train_score)+1)],train_score,label="train")
    plt.plot([i for i in range(1,len(test_score)+1)],test_score,label="test")
    
    plt.legend()
    plt.show()
